def Norman(x, y, mu, A, L, M=None, c=None):
    if M:
        Gamma = A + M + "^T(" + L + "+" + M + A + M + "^T)^{-1}"
        xmu   = mu + "+" + Gamma + "(" + y + "-" + M + mu #need to add c
        xvar  = A + "-" + Gamma + M + A
        ymu   = M + mu #need to add c
        yvar  = L + "+" + M + A + M + "^T"
        newC = mu + "-" + Gamma + "(" + M + mu
    else:
        Gamma = A + "(" + L + "+" +  A + ")^{-1}"
        xmu   = mu + "+" + Gamma + "(" + y + "-" + mu #need to add c
        xvar  = A + "-" + Gamma +  A
        ymu   = mu #need to add c
        yvar  = L + "+" + A
        newC = mu + "-" + Gamma + "(" + mu
    if c: 
        xmu = xmu + "-" + c + ")"
        ymu = ymu + "+" + c + ")"
        newC = newC + "-" + c + ")"
    else: 
        ymu = ymu
        xmu = xmu + ")"
        newC = newC + ")"
    newM = Gamma
    
    string1 = "\mathcal{N} ( " + y + " \mid " + ymu + "," + yvar + ")"
    string2 =  "\mathcal{N} ( " + x + " \mid " + xmu + "," + xvar + ")" 
    return(string1 ,string2, newM, newC)
